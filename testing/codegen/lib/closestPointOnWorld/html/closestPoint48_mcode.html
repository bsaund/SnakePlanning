1function p_plane = closestPoint(p_test, p1, p2, p3)
  2%CLOSESTPOINT returns the closest points to p_test on the triangle
  3%defined by p1, p2, and p3. All points are column vectors.
  4%This uses the algorithm from:
  5%http://www.geometrictools.com/Documentation/DistancePoint3Triangle3.pdf
  6
  7    B = p1;
  8    E0 = p2-p1;
  9    E1 = p3-p1;
 10    P = p_test;
 11    D = B-P;
 12%     a = dot(E0,E0);
 13%     b = dot(E0,E1);
 14%     c = dot(E1,E1);
 15%     d = dot(E0,D);
 16%     e = dot(E1,D);
 17    a = E0'*E0;
 18    b = E0'*E1;
 19    c = E1'*E1;
 20    d = E0'*D;
 21    e = E1'*D;
 22%     f = dot(D,D);
 23    
 24    det = a*c-b*b;
 25    s = b*e-c*d;
 26    t = b*d-a*e;
 27  
 28 
 29    
 30    
 31    if(s+t < det)
 32        if (s<0)
 33            if(t<0)
 34                % region = 4
 35                if (d<0)
 36                    s = bound(-d/a, 0, 1);
 37                    t = 0;
 38                else
 39                    s = 0;
 40                    t = bound(-e/c, 0, 1);
 41                end
 42                
 43            else
 44                % region = 3
 45                s = 0;
 46                t = bound(-e/c, 0, 1);
 47            end
 48        elseif (t<0)
 49            % region = 5
 50            s = bound(-d/a, 0, 1);
 51            t = 0;
 52        else
 53            % region = 0
 54            invDet = 1/det;
 55            s = s*invDet;
 56            t = t*invDet;
 57        end
 58    else
 59        if (s<0)
 60            % region = 2
 61            tmp0 = b+d;
 62            tmp1 = c+e;
 63            if (tmp1 > tmp0)
 64                numer = tmp1-tmp0;
 65                denom = a-2*b+c;
 66                s = bound(numer/denom, 0, 1);
 67                t = 1-s;
 68            else
 69                s=0;
 70                t = bound(-e/c, 0, 1);
 71            end
 72
 73        elseif (t<0)
 74            
 75            % region = 6
 76            
 77            tmp0 = b+e;
 78            tmp1 = a+d;
 79            if (tmp1 > tmp0)
 80                numer = tmp1-tmp0;
 81                denom = a-2*b+c;
 82                t = bound(numer/denom, 0, 1);
 83                s = 1-t;
 84            else
 85                t=0;
 86                s = bound(-d/a, 0, 1);
 87            end
 88
 89        else 
 90            % region = 1
 91            numer = c+e-b-d;
 92            denom = a-2*b+c;
 93            s = bound(numer/denom, 0, 1);
 94            t = 1-s;
 95        end
 96    end
 97       
 98    p_plane = B+E0*s + E1*t;
 99end
