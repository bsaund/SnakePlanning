1function p = closestPointOnWorld(p_test, world)
 2%CLOSESTPOINTONWORLD finds the closest point to p_test that lies on
 3%the mesh 'world'. world should be a struct of faces and vertices,
 4%the same struct you might pass to matlab's 'patch'
 5
 6    f = world.faces;
 7    v = world.vertices;
 8    numFaces = size(f,1);
 9
10    
11    d = inf;
12    p = [0;0;0];
13    for i=1:numFaces
14        p_temp = closestPoint(p_test, ...
15            v(f(i,1),:)',...
16            v(f(i,2),:)',...
17            v(f(i,3),:)');
18        d_temp = (p_test-p_temp)'*(p_test - p_temp);
19
20        if (d_temp < d)
21            d = d_temp;
22            p = p_temp;
23        end
24    end
25                                      
26end
